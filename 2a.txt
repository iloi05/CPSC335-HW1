#include  <iostream>
#include <cmath>

int squareRootCompute(number) {
    int root = 0;
    //Checking if square root is radical
    if (root^2 != number) then:
        print out "The number inputted is a radical";
        return 1
    // Checking if square root is imaginary
    else if (number < 0) then:
        print out "The number is negative, cannot compute square root because root is imaginary";
        return 1;
    else
        root * root == number;
        print out "The root of (number) is (root)";
    
    return root;
}
